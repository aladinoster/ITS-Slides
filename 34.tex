    \begin{exampleblock}{Optimization}
    In order to optimize the \emph{cost} $f(x)$, the value of $x$ should be found, while respecting the constraints
        \begin{equation*}
        x^*=
        \begin{aligned}
        &\underset{x}{\min}& &f(x)\\
        &\text{s.t}& &x\geq0.5
        \end{aligned}
        \end{equation*}
    \end{exampleblock}
    }
